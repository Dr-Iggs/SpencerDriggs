library(tidyverse)
library(USAboundaries)
library(USAboundariesData)
library(sf)
library(pander)
library(ggrepel)
library(leaflet)

states <- (us_states(resolution='low')) %>%
  select(name, geometry) %>%
  filter(!name %in% c('Alaska','Puerto Rico','Hawaii','District of Columbia'))
county <- us_counties(states='ID')
top3 <- us_cities() %>%
  filter(!state_name %in% c('Alaska','Puerto Rico','Hawaii','District of Columbia')) %>%
  arrange(desc(population)) %>%
  group_by(state_name) %>%
  slice_head(n=3) %>%
  mutate(cityscore=0)

top3$cityscore = rep(c(1,2,3),length(top3$cityscore)/3)
Code 
ggplot() +
  geom_sf(data = states,fill='white') +
  geom_sf(data = county,fill='white') +
  geom_sf(data=top3, aes(size=population/1000,color=cityscore)) +
  #scale_fill_manual(values=c('1'='navyblue','2'='blue','3'='lightblue')) +
  geom_label_repel(data=top3,aes(label=ifelse(cityscore==1,city,element_blank()), geometry = geometry), stat = "sf_coordinates", size = 1) +
  theme_bw()
